import { supabase } from "./supabase.js"; const bikeSelect = document.getElementById("bikeSelect"); const saveBtn = document.getElementById("saveBtn"); let bikes = []; let currentBike = null; async function loadBikes() { const { data, error } = await supabase .from("motorcycles") .select("*") .order("model", { ascending: true }); if (error) { console.error(error); alert("Failed to load bikes"); return; } bikes = data; bikeSelect.innerHTML = bikes .map(b => ``) .join(""); if (bikes.length) { loadBike(bikes[0].slug); } } async function loadBike(slug) { const { data, error } = await supabase .from("motorcycles") .select("*") .eq("slug", slug) .single(); if (error) { console.error(error); return; } currentBike = data; document.getElementById("model").value = data.model || ""; document.getElementById("category").value = data.category || ""; document.getElementById("unitPrice").value = data.unit_price || ""; } async function saveBike() { if (!currentBike) return; const updated = { model: document.getElementById("model").value, category: document.getElementById("category").value, unit_price: document.getElementById("unitPrice").value }; const { error } = await supabase .from("motorcycles") .update(updated) .eq("slug", currentBike.slug); if (error) { console.error(error); alert("Failed to save"); return; } alert("Saved successfully"); loadBikes(); } bikeSelect.addEventListener("change", (e) => { loadBike(e.target.value); }); saveBtn.addEventListener("click", saveBike); loadBikes();
